  }
}

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define ANCHO_PANTALLA 128
#define ALTO_PANTALLA 64
#define OLED_ADDR 0x3C

Adafruit_SSD1306 display(ANCHO_PANTALLA, ALTO_PANTALLA, &Wire);

int estado = 0; // 0 = feliz ojos abiertos, 1 = feliz ojos cerrados, 2 = triste ojos abiertos

// Función para dibujar arco (aproximación simple)
void drawArc(int cx, int cy, int rx, int ry, int startAngle, int endAngle, uint16_t color) {
  for (int angle = startAngle; angle <= endAngle; angle++) {
    float rad = angle * 3.14159 / 180.0;
    int x = cx + cos(rad) * rx;
    int y = cy + sin(rad) * ry;
    display.drawPixel(x, y, color);
  }
}

void dibujarCarita(int tipo) {
  int centroX = ANCHO_PANTALLA / 2;
  int centroY = ALTO_PANTALLA / 2;
  int radio = 25;

  display.clearDisplay();

  // Círculo externo
  display.drawCircle(centroX, centroY, radio, SSD1306_WHITE);

  // Ojos
  if (tipo == 0) {
    // Feliz ojos abiertos
    display.fillCircle(centroX - 10, centroY - 5, 4, SSD1306_WHITE);
    display.fillCircle(centroX + 10, centroY - 5, 4, SSD1306_WHITE);
  } else if (tipo == 1) {
    // Feliz ojos cerrados (líneas)
    display.drawLine(centroX - 14, centroY - 5, centroX - 6, centroY - 5, SSD1306_WHITE);
    display.drawLine(centroX + 6, centroY - 5, centroX + 14, centroY - 5, SSD1306_WHITE);
  } else if (tipo == 2) {
    // Triste ojos abiertos (igual que feliz)
    display.fillCircle(centroX - 10, centroY - 5, 4, SSD1306_WHITE);
    display.fillCircle(centroX + 10, centroY - 5, 4, SSD1306_WHITE);
  }

  // Boca
  if (tipo == 0 || tipo == 1) {
    // Boca feliz, arco de 20 a 160 grados
    drawArc(centroX, centroY + 10, 15, 10, 20, 160, SSD1306_WHITE);
  } else if (tipo == 2) {
    // Boca triste, arco de 200 a 340 grados
    drawArc(centroX, centroY + 20, 15, 10, 200, 340, SSD1306_WHITE);
  }

  display.display();
}

void setup() {
  Serial.begin(9600);
  if(!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR)) {
    Serial.println(F("No se encontró pantalla SSD1306"));
    while(true);
  }
  display.clearDisplay();
  delay(500);
}

void loop() {
  switch (estado) {
    case 0:
      dibujarCarita(0);  // feliz ojos abiertos
      delay(1000);
      estado = 1;
      break;
    case 1:
      dibujarCarita(1);  // feliz ojos cerrados
      delay(300);
      estado = 2;
      break;
    case 2:
      dibujarCarita(2);  // triste ojos abiertos
      delay(1500);
      estado = 0;
      break;
  }
}
